  핵심 데이터 흐름:
  1. Admin이 캐릭터 생성 요청 → Claude API가 페르소나 JSON 생성
  2. 페르소나 기반으로 이미지 생성 job 큐잉 → Image Provider가 생성
  3. Safety Layer 통과 → 실패 시 재생성 또는 reject
  4. 승인된 미디어 → Publishing Service가 Instagram Graph API로 게시

  Provider Abstraction 상세:

  // src/services/media/providers/types.ts

  export interface ImageGenerationInput {
    prompt: string;
    negativePrompt: string;
    width: number;
    height: number;
    seed?: number;
  }

  export interface ImageGenerationOutput {
    url: string;          // 생성된 이미지 URL (provider 측)
    seed: number;
    metadata: Record<string, unknown>;
  }

  export interface ImageProvider {
    readonly name: string;
    generate(input: ImageGenerationInput): Promise<ImageGenerationOutput>;
    checkStatus(jobId: string): Promise<'pending' | 'completed' | 'failed'>;
  }

  export interface VideoGenerationInput {
    sourceImageUrl: string;   // 기반 이미지
    motionPrompt: string;     // 움직임 설명
    duration: 3 | 5 | 10;    // 초
  }

  export interface VideoGenerationOutput {
    url: string;
    durationMs: number;
    metadata: Record<string, unknown>;
  }

  export interface VideoProvider {
    readonly name: string;
    generate(input: VideoGenerationInput): Promise<VideoGenerationOutput>;
    checkStatus(jobId: string): Promise<'pending' | 'completed' | 'failed'>;
  }

  4. DB 테이블 설계

  -- 캐릭터 테이블
  CREATE TABLE characters (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      name            VARCHAR(100) NOT NULL,
      slug            VARCHAR(100) UNIQUE NOT NULL,
      persona         JSONB NOT NULL,
      -- persona 예시: {
      --   "age": 25,
      --   "gender": "female",
      --   "nationality": "Korean",
      --   "occupation": "digital artist",
      --   "personality": ["cheerful", "creative"],
      --   "visual_description": "long black hair, warm brown eyes...",
      --   "voice_tone": "friendly and witty",
      --   "hashtag_style": ["#art", "#digitalart"],
      --   "banned_topics": ["politics", "religion"]
      -- }
      status          VARCHAR(20) NOT NULL DEFAULT 'draft',
      -- draft | active | paused | archived
      ig_account_id   VARCHAR(100),         -- 연결된 IG 계정
      created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
      updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

  -- 미디어 에셋 테이블
  CREATE TABLE media_assets (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      character_id    UUID NOT NULL REFERENCES characters(id),
      type            VARCHAR(10) NOT NULL,  -- image | video
      provider        VARCHAR(50) NOT NULL,  -- replicate | fal | runway
      prompt          TEXT NOT NULL,
      negative_prompt TEXT,
      storage_key     VARCHAR(500),          -- S3 key
      storage_url     VARCHAR(1000),         -- CDN URL
      width           INT,
      height          INT,
      duration_ms     INT,                   -- 영상일 경우
      seed            BIGINT,
      generation_params JSONB,               -- provider별 추가 파라미터
      safety_status   VARCHAR(20) NOT NULL DEFAULT 'pending',
      -- pending | passed | failed | manual_review
      safety_result   JSONB,                 -- 검증 상세 결과
      status          VARCHAR(20) NOT NULL DEFAULT 'generating',
      -- generating | ready | published | rejected | error
      created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

  CREATE INDEX idx_media_character ON media_assets(character_id);
  CREATE INDEX idx_media_status ON media_assets(status);

  -- 게시 테이블
  CREATE TABLE posts (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      character_id    UUID NOT NULL REFERENCES characters(id),
      media_asset_id  UUID NOT NULL REFERENCES media_assets(id),
      caption         TEXT NOT NULL,
      hashtags        TEXT[] NOT NULL,
      post_type       VARCHAR(20) NOT NULL,  -- feed | reel | story | carousel
      ig_media_id     VARCHAR(100),          -- IG 게시 후 반환 ID
      ig_permalink    VARCHAR(500),
      scheduled_at    TIMESTAMPTZ,           -- 예약 게시 시간
      published_at    TIMESTAMPTZ,
      status          VARCHAR(20) NOT NULL DEFAULT 'draft',
      -- draft | scheduled | publishing | published | failed
      engagement      JSONB,                 -- { likes, comments, reach, 
  impressions }
      created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

  CREATE INDEX idx_posts_scheduled ON posts(scheduled_at) WHERE status =
  'scheduled';

  -- Job 추적 테이블
  CREATE TABLE jobs (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      queue_name      VARCHAR(50) NOT NULL,
      bull_job_id     VARCHAR(100) NOT NULL,
      entity_type     VARCHAR(30) NOT NULL,  -- character | media_asset | post
      entity_id       UUID NOT NULL,
      status          VARCHAR(20) NOT NULL DEFAULT 'waiting',
      -- waiting | active | completed | failed | dead
      attempts        INT NOT NULL DEFAULT 0,
      max_attempts    INT NOT NULL DEFAULT 3,
      error_message   TEXT,
      started_at      TIMESTAMPTZ,
      completed_at    TIMESTAMPTZ,
      created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

  CREATE INDEX idx_jobs_entity ON jobs(entity_type, entity_id);
  CREATE INDEX idx_jobs_status ON jobs(status);

  -- Safety 감사 로그
  CREATE TABLE safety_audit_log (
      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      media_asset_id  UUID REFERENCES media_assets(id),
      post_id         UUID REFERENCES posts(id),
      check_type      VARCHAR(30) NOT NULL,  -- nsfw | age | brand_safety
      input_type      VARCHAR(10) NOT NULL,  -- text | image | video
      result          VARCHAR(10) NOT NULL,  -- pass | fail | warn
      score           DECIMAL(5,4),          -- 0.0000 ~ 1.0000
      details         JSONB,
      created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
  );

  Job 상태 전이

  waiting → active → completed
                   → failed → (retry) → active
                            → (max retries) → dead

  각 worker는 DB의 jobs 테이블과 해당 entity 테이블 상태를 동기화:

  // src/jobs/workers/imageWorker.ts (핵심 로직)
  import { Worker } from 'bullmq';
  import { db } from '../../db/client';
  import { mediaAssets, jobs } from '../../db/schema';
  import { providerFactory } from '../../services/media/providerFactory';
  import { safetyGate } from '../../safety/safetyGate';
  import { storageService } from '../../services/storage/storageService';

  const imageWorker = new Worker('image', async (job) => {
    const { mediaAssetId, characterId, prompt, negativePrompt, provider:
  providerName } = job.data;

    // 1. 이미지 생성
    const provider = providerFactory.getImageProvider(providerName);
    const result = await provider.generate({
      prompt,
      negativePrompt,
      width: 1080,
      height: 1350, // IG portrait 비율
    });

    // 2. S3 업로드
    const storageKey = `characters/${characterId}/images/${mediaAssetId}.png`;
    const storageUrl = await storageService.uploadFromUrl(result.url,
  storageKey);

    // 3. DB 업데이트
    await db.update(mediaAssets)
      .set({ storageKey, storageUrl, seed: result.seed, status: 'ready' })
      .where(eq(mediaAssets.id, mediaAssetId));

    // 4. Safety 검증 job 큐잉
    await safetyQueue.add('check', {
      mediaAssetId,
      storageUrl,
      type: 'image',
    });

    return { mediaAssetId, storageUrl };
  });

  async function generateImageWithFallback(input: ImageGenerationInput): 
  Promise<ImageGenerationOutput> {
    const providers = providerFactory.getImageProviderChain(); // [replicate, 
  fal]

    for (const provider of providers) {
      try {
        return await provider.generate(input);
      } catch (err) {
        logger.warn({ provider: provider.name, err }, 'Provider failed, trying 
  next');
        if (provider === providers.at(-1)) throw err; // 마지막 provider면 throw
      }
    }
    throw new Error('All providers exhausted');
  }

  imageWorker.on('failed', async (job, err) => {
    if (job && job.attemptsMade >= job.opts.attempts!) {

      await db.update(jobs)

        queue: 'image',
        jobId: job.id,
        entityId: job.data.mediaAssetId,
        error: err.message,
      });
    }

  export class SafetyGate {
    async validateImage(imageUrl: string): Promise<SafetyResult> {
      const results = await Promise.all([
        this.nsfwFilter.checkImage(imageUrl),
        this.ageGuard.checkImage(imageUrl),
        this.brandSafety.checkImage(imageUrl),

      // 하나라도 fail이면 전체 fail

      if (failed.length > 0) {

  -- Instagram 게시 준비

      id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),

      media_ids       UUID[] NOT NULL,       -- images.id 또는 videos.id 배열

      hashtags        TEXT[] DEFAULT '{}',

      created_at      TIMESTAMPTZ NOT NULL DEFAULT now()

  CREATE INDEX idx_posts_scheduled ON posts(scheduled_at) WHERE status =
  'scheduled';

      id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  CREATE INDEX idx_moderation_target ON moderation_logs(target_type, target_id);

      finished_at TIMESTAMPTZ,

  큐 구성 (BullMQ)

  redisConnection });

  redisConnection });

  전체 파이프라인 흐름

  [1] character-generation
      Input:  { theme, style, constraints }
      Action: Claude API → persona + visual_prompt 생성
      Output: characters 테이블 INSERT
      Next:   → image-generation (자동 체이닝)

  [2] image-generation
      Input:  { characterId, scenePrompt?, provider }
      Action: PromptSanitizer → Provider.generate() → 파일 저장
      Output: images 테이블 UPDATE (storage_url)
      Next:   → moderation

  [3] video-generation
      Input:  { imageId, motionPrompt, provider }
      Action: PromptSanitizer → Provider.generate() → 폴링 → 파일 저장
      Output: videos 테이블 UPDATE (storage_url)
      Next:   → moderation

  [4] moderation
      Input:  { targetType, targetId }
      Action: NsfwFilter + AgeGateValidator + BrandSafetyChecker
      Output: moderation_logs INSERT, 대상 status → approved/rejected
      Next:   rejected → 알림 / approved → 게시 가능 상태

  [5] publish
      Input:  { postId }
      Action: Instagram Graph API 호출
      Output: posts 테이블 UPDATE (ig_post_id, published_at)

  Job 옵션 기본값

  // 각 큐별 기본 설정
  const defaultJobOptions = {
    'character-generation': {
      attempts: 3,
      backoff: { type: 'exponential', delay: 5000 },  // 5s, 10s, 20s
      removeOnComplete: { age: 86400 },  // 24h 후 정리
      removeOnFail: { age: 604800 },     // 7일 보관
    },
    'image-generation': {
      attempts: 2,
      backoff: { type: 'fixed', delay: 10000 },
      timeout: 120000,  // 2분
    },
    'video-generation': {
      attempts: 2,
      backoff: { type: 'fixed', delay: 30000 },
      timeout: 600000,  // 10분 (영상 생성은 오래 걸림)
    },
    'moderation': {
      attempts: 3,
      backoff: { type: 'exponential', delay: 2000 },
      timeout: 30000,
    },
    'publish': {
      attempts: 5,
      backoff: { type: 'exponential', delay: 60000 },  // rate limit 대응
      timeout: 30000,
    },
  };

  Job 상태 전이도

  WAITING → ACTIVE → COMPLETED
                   ↘ FAILED → (auto retry) → ACTIVE
                            → (max attempts) → PERMANENTLY_FAILED → 수동 개입

  ┌──────────────┬───────────────────┬──────────────────────────────────────┐

  ├──────────────┼───────────────────┼──────────────────────────────────────┤
  │ Video        │ 생성 실패 / 폴링  │ 10분 timeout. 실패 시 원본 이미지    │
  │ Provider     │ timeout           │ 유지, 재시도 가능                    │
  ├──────────────┼───────────────────┼──────────────────────────────────────┤
  │ Moderation   │ NSFW / 미성년     │ 자동 재생성 안 함. rejected 상태로   │
  │ 거부         │ 암시 탐지         │ 기록. 관리자 확인 필요               │
  ├──────────────┼───────────────────┼──────────────────────────────────────┤
  │ Storage      │ S3 오류           │ 3회 재시도. 실패 시 Job failed,      │
  │ 업로드       │                   │ 생성물 로컬 임시 보관                │
  ├──────────────┼───────────────────┼──────────────────────────────────────┤
  │ Instagram    │ Rate limit /      │ Exponential backoff. token 만료 시   │
  │ API          │ token 만료        │ 알림 발송                            │
  ├──────────────┼───────────────────┼──────────────────────────────────────┤
  │ DB 연결      │ Connection pool   │ Graceful degradation: 큐 일시 정지,  │
  │              │ 소진              │ 연결 복구 후 재개                    │
  └──────────────┴───────────────────┴──────────────────────────────────────┘

  Provider Fallback 체인

  // src/services/ImageService.ts
  export class ImageService {
    private providers: ImageProvider[];

    async generateWithFallback(req: ImageGenerationRequest):
  Promise<ImageGenerationResult> {
      for (const provider of this.providers) {
        try {
          return await provider.generate(req);
        } catch (err) {
          logger.warn({ provider: provider.name, err }, 'Provider failed, trying
   next');
          continue;
        }
      }
      throw new AllProvidersFailedError();
    }
  }

  // 최종 실패 Job은 별도 DLQ로 이동
  imageQueue.on('failed', async (job, err) => {
    if (job.attemptsMade >= job.opts.attempts) {

  export class AgeGateValidator {

    validateCharacterPersona(persona: CharacterPersona): ValidationResult {
      const violations: string[] = [];

      if (persona.age < 22) {
        violations.push(`Age ${persona.age} is below minimum 22`);
      }

      const blockedTerms = ['teen', 'student', 'school', 'juvenile',

      for (const term of blockedTerms) {

          violations.push(`Blocked term found: "${term}"`);
        }
      }

      return { passed: violations.length === 0, violations };

     - @anthropic-ai/sdk 연동

  2. Safety 기본 구현
